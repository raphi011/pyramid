{{- if .Values.ingress.enabled }}

Access the application at:
  https://{{ .Values.ingress.host }}
{{- else }}

Get the application URL by running:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "pyramid.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward $POD_NAME 3000:3000
  echo "Visit http://127.0.0.1:3000"
{{- end }}

Don't forget to run database migrations:
  kubectl exec -it deploy/{{ include "pyramid.fullname" . }} -- bun run db:migrate
